#### 1.DeviceEventEmitter
```js
//注册这个监听事件
componentDidMount(){
  this.listener=DeviceEventEmitter.addListener('xxxName',()=>{});
};
//在组件销毁的时候要将其移除
componentWillUnmount(){
    this.listener.remove();
};
//发送通知
DeviceEventEmitter.emit('xxxName','通知来了');
```
